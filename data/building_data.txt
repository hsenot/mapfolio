Getting building data:

Create a structure to host the building data:
CREATE TABLE osm_buildings
(
  osm_id integer NOT NULL,
  building text,
  name text,
  addr_housenumber character varying,
  addr_street character varying,
  addr_city character varying,
  addr_postcode character varying,
  the_geom geometry(Geometry,4326),
  CONSTRAINT osm_building_pk PRIMARY KEY (osm_id )
)

Download the buildings data file from:
http://app.carbongis.com.au/bip/data/PGDUMP/buildings.sql.zip

Unzip

Load the data into the local structure:
copy osm_buildings from '/path/to/file';


-- danger zone: make sure this table does not already contain user-entered data!
drop table building cascade;

CREATE TABLE building
(
  id serial,
  name text,
  source text,
  source_id integer,
  the_geom geometry(Geometry,4326),
  CONSTRAINT building_pk PRIMARY KEY (id)
);

CREATE INDEX building_geom_idx ON building USING gist (the_geom);

insert into building (name,source,source_id,the_geom)
select 
coalesce(name,case building when 'yes' then '' else building end),
cast('OSM' as text),
osm_id,
the_geom
from osm_buildings;
